# 1. Copy Input file from source to binary dir
configure_file(${CMAKE_CURRENT_SOURCE_DIR}/inputT.yaml
               ${CMAKE_CURRENT_BINARY_DIR}/inputT.yaml COPYONLY)
configure_file(${CMAKE_CURRENT_SOURCE_DIR}/materials.yaml
               ${CMAKE_CURRENT_BINARY_DIR}/materials.yaml COPYONLY)
configure_file(${CMAKE_CURRENT_SOURCE_DIR}/assoc.txt
               ${CMAKE_CURRENT_BINARY_DIR}/assoc.txt COPYONLY)
configure_file(${CMAKE_CURRENT_SOURCE_DIR}/inputVerificationT.yaml
               ${CMAKE_CURRENT_BINARY_DIR}/inputVerificationT.yaml COPYONLY)

# 1. Name the test with the directory name, prepended by SCOREC_
get_filename_component(testRoot ${CMAKE_CURRENT_SOURCE_DIR} NAME)
set(testName "SCOREC_${testRoot}")

# 1. Create the test with this name and standard executable
if(ALBANY_IFPACK2)
  add_test(NAME ${testName}_Tpetra COMMAND ${Albany.exe} inputT.yaml)
  set_tests_properties(${testName}_Tpetra
                       PROPERTIES LABELS "LCM;Tpetra;Forward;SCOREC")
  add_test(NAME ${testName}_Verification_Tpetra COMMAND ${Albany.exe}
                                                        inputVerificationT.yaml)
  set_tests_properties(${testName}_Verification_Tpetra
                       PROPERTIES LABELS "LCM;Tpetra;Forward;SCOREC")
endif()
