if(ALBANY_DTK)
  # 1. Create the test with this name and standard executable
  get_filename_component(testName ${CMAKE_CURRENT_SOURCE_DIR} NAME)
  set(OUTFILE "Parallel_Cubes_4_DBC.Log")
  set(PYTHON_FILE "check_comparisons_dbc.py")
  add_test(
    NAME Schwarz_${testName}_DTK_DBC_np4
    COMMAND
      ${CMAKE_COMMAND} "-DTEST_PROG=${Albany.exe}" -DTEST_NAME=Cubes
      -DTEST_ARGS=cubes.yaml -DMPIMNP=4 -DLOGFILE=${OUTFILE}
      -DPY_FILE=${PYTHON_FILE} -DDATA_DIR=${CMAKE_CURRENT_SOURCE_DIR} -P
      ${runtest.cmake})
  set_tests_properties(Schwarz_${testName}_DTK_DBC_np4
                       PROPERTIES LABELS "LCM;Tpetra;Forward")
  set(OUTFILE "Parallel_Cubes_4_SDBC.Log")
  set(PYTHON_FILE "check_comparisons_sdbc.py")
  add_test(
    NAME Schwarz_${testName}_DTK_SDBC_np4
    COMMAND
      ${CMAKE_COMMAND} "-DTEST_PROG=${Albany.exe}" -DTEST_NAME=CubesSDBC
      -DTEST_ARGS=cubes_sdbc.yaml -DMPIMNP=4 -DLOGFILE=${OUTFILE}
      -DPY_FILE=${PYTHON_FILE} -DDATA_DIR=${CMAKE_CURRENT_SOURCE_DIR} -P
      ${runtest.cmake})
  set_tests_properties(Schwarz_${testName}_DTK_SDBC_np4
                       PROPERTIES LABELS "LCM;Tpetra;Forward")
endif()
